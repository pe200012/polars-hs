# Arrow C Data Interface Import Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `fromArrowRecordBatch` for importing a standard Arrow C Data Interface RecordBatch into a managed Polars `DataFrame`.

**Architecture:** Haskell exposes a small `Polars.Arrow` boundary with an explicit `unsafeArrowRecordBatch` pointer wrapper and a safe `fromArrowRecordBatch` result type. Rust consumes a top-level struct `ArrowSchema` + struct `ArrowArray`, imports them through `polars-arrow`, splits the imported `StructArray` into child arrays, builds Polars Series, and assembles a DataFrame through `DataFrame::new_infer_height`.

**Tech Stack:** Haskell Stack/GHC 9.12.2, Hspec, Rust 2024, Polars 0.53.0, `polars-arrow` 0.53.0, cbindgen, Apache Arrow C Data Interface.

```haskell
releaseSchemaImpl :: Ptr ArrowSchema -> IO ()

    rawPrivate <- readAt ptr schemaPrivateDataOffset :: IO (Ptr ())

    freeStablePtr stable

withPeopleRecordBatch :: (Ptr ArrowSchema -> Ptr ArrowArray -> IO a) -> IO a
withPeopleRecordBatch action = do
    nameSchema <- makeLeafSchema "u" "name" True
    ageSchema <- makeLeafSchema "l" "age" True
    nameArray <- makeUtf8Array
    ageArray <- makeInt64Array
    topSchema <- makeStructSchema [nameSchema, ageSchema]
    topArray <- makeStructArray 3 [nameArray, ageArray]

makeLeafSchema :: String -> String -> Bool -> IO (Ptr ArrowSchema)
makeLeafSchema format name nullable = do
    ptr <- mallocBytes arrowSchemaSize
    formatPtr <- newCString format
    namePtr <- newCString name
    ptr `at` schemaFormatOffset $ formatPtr
    ptr `at` schemaNameOffset $ namePtr
    ptr `at` schemaMetadataOffset $ (nullPtr :: Ptr ())
    ptr `at` schemaFlagsOffset $ arrowFlags nullable
    ptr `at` schemaNChildrenOffset $ (0 :: Int64)
    ptr `at` schemaChildrenOffset $ (nullPtr :: Ptr (Ptr ArrowSchema))
    ptr `at` schemaDictionaryOffset $ (nullPtr :: Ptr ArrowSchema)
    ptr `at` schemaReleaseOffset $ releaseSchemaPtr
    cleanup <- newStablePtr $ free formatPtr >> free namePtr >> free ptr
    ptr `at` schemaPrivateDataOffset $ castStablePtrToPtr cleanup
    pure ptr

makeStructSchema :: [Ptr ArrowSchema] -> IO (Ptr ArrowSchema)
makeStructSchema children = do
    ptr <- mallocBytes arrowSchemaSize
    formatPtr <- newCString "+s"
    namePtr <- newCString ""
    childrenPtr <- newArray children
    ptr `at` schemaFormatOffset $ formatPtr
    ptr `at` schemaNameOffset $ namePtr
    ptr `at` schemaMetadataOffset $ (nullPtr :: Ptr ())
    ptr `at` schemaFlagsOffset $ arrowFlags False
    ptr `at` schemaNChildrenOffset $ (fromIntegral (length children) :: Int64)
    ptr `at` schemaChildrenOffset $ childrenPtr
    ptr `at` schemaDictionaryOffset $ (nullPtr :: Ptr ArrowSchema)
    ptr `at` schemaReleaseOffset $ releaseSchemaPtr

        forM_ children releaseIfLiveSchema
        free childrenPtr
        free formatPtr
        free namePtr
        free ptr

makeLeafArray :: Int -> Int64 -> [Ptr ()] -> IO () -> IO (Ptr ArrowArray)
makeLeafArray len nullCount buffers cleanupBuffers = do
    ptr <- mallocBytes arrowArraySize
    buffersPtr <- mallocArray (length buffers) :: IO (Ptr (Ptr ()))

    ptr `at` arrayLengthOffset $ (fromIntegral len :: Int64)
    ptr `at` arrayNullCountOffset $ nullCount
    ptr `at` arrayOffsetOffset $ (0 :: Int64)
    ptr `at` arrayNBuffersOffset $ (fromIntegral (length buffers) :: Int64)
    ptr `at` arrayNChildrenOffset $ (0 :: Int64)
    ptr `at` arrayBuffersOffset $ buffersPtr
    ptr `at` arrayChildrenOffset $ (nullPtr :: Ptr (Ptr ArrowArray))
    ptr `at` arrayDictionaryOffset $ (nullPtr :: Ptr ArrowArray)
    ptr `at` arrayReleaseOffset $ releaseArrayPtr
    cleanup <- newStablePtr $ cleanupBuffers >> free buffersPtr >> free ptr
    ptr `at` arrayPrivateDataOffset $ castStablePtrToPtr cleanup
    pure ptr

makeStructArray len children = do
    ptr <- mallocBytes arrowArraySize
    buffersPtr <- newArray [nullPtr :: Ptr ()]
    childrenPtr <- newArray children
    ptr `at` arrayLengthOffset $ (fromIntegral len :: Int64)
    ptr `at` arrayNullCountOffset $ (0 :: Int64)
    ptr `at` arrayOffsetOffset $ (0 :: Int64)
    ptr `at` arrayNBuffersOffset $ (1 :: Int64)
    ptr `at` arrayNChildrenOffset $ (fromIntegral (length children) :: Int64)
    ptr `at` arrayBuffersOffset $ buffersPtr
    ptr `at` arrayChildrenOffset $ childrenPtr
    ptr `at` arrayDictionaryOffset $ (nullPtr :: Ptr ArrowArray)
    ptr `at` arrayReleaseOffset $ releaseArrayPtr
    cleanup <- newStablePtr $ do
        forM_ children releaseIfLiveArray
        free childrenPtr
        free buffersPtr
        free ptr
    ptr `at` arrayPrivateDataOffset $ castStablePtrToPtr cleanup
    pure ptr
```

```rust
use std::ffi::c_void;
use std::os::raw::c_int;
use std::ptr;

use polars::prelude::*;
use polars_arrow::array::StructArray;
use polars_arrow::datatypes::ArrowDataType;
use polars_arrow::ffi::{ArrowArray, ArrowSchema, import_array_from_c, import_field_from_c};

use crate::error::{PhsError, PhsResult, ffi_boundary, phs_error, required_mut};
use crate::handles::{dataframe_into_raw, phs_dataframe};

#[unsafe(no_mangle)]
pub unsafe extern "C" fn phs_dataframe_from_arrow_record_batch(
    schema: *mut c_void,
    array: *mut c_void,
    out: *mut *mut phs_dataframe,
    err: *mut *mut phs_error,
) -> c_int {
    ffi_boundary(err, || {
        let out = unsafe { required_mut(out, "out") }?;
        *out = ptr::null_mut();
        let schema = required_arrow_ptr::<ArrowSchema>(schema, "schema")?;
        let array = required_arrow_ptr::<ArrowArray>(array, "array")?;
        unsafe { ensure_schema_live(schema)? };
        unsafe { ensure_array_live(array)? };

        let field_result = unsafe { import_field_from_c(&*schema) };
        unsafe { release_schema(schema) };
        let field = field_result?;
        if !matches!(&field.dtype, ArrowDataType::Struct(_)) {
            unsafe { release_array(array) };
            return Err(PhsError::invalid_argument("Arrow RecordBatch schema must be a struct"));
        }

        let array_value = unsafe { take_array(array)? };
        let imported = unsafe { import_array_from_c(array_value, field.dtype.clone()) }?;
        let struct_array = imported
            .as_any()
            .downcast_ref::<StructArray>()
            .ok_or_else(|| PhsError::invalid_argument("Arrow RecordBatch array must be a struct"))?
            .clone();
        let (fields, _, arrays, validity) = struct_array.into_data();
        if validity.as_ref().is_some_and(|bitmap| bitmap.unset_bits() > 0) {
            return Err(PhsError::invalid_argument("Arrow RecordBatch struct nulls are unsupported"));
        }

        let mut columns = Vec::with_capacity(fields.len());
        for (field, array) in fields.into_iter().zip(arrays) {
            let series = unsafe {
                Series::_try_from_arrow_unchecked_with_md(
                    field.name.clone(),
                    vec![array],
                    field.dtype(),
                    field.metadata.as_deref(),
                )?
            };
            columns.push(series.into());
        }
        *out = dataframe_into_raw(DataFrame::new_infer_height(columns)?);
        Ok(())
    })
}

fn required_arrow_ptr<T>(ptr: *mut c_void, name: &str) -> PhsResult<*mut T> {
    if ptr.is_null() {
        Err(PhsError::invalid_argument(format!("{name} pointer was null")))
    } else {
        Ok(ptr.cast::<T>())
    }
}

unsafe fn ensure_schema_live(schema: *mut ArrowSchema) -> PhsResult<()> {
    if unsafe { (*schema).release.is_none() } {
        Err(PhsError::invalid_argument("schema was already released"))
    } else {
        Ok(())
    }
}

unsafe fn ensure_array_live(array: *mut ArrowArray) -> PhsResult<()> {
    if unsafe { (*array).release.is_none() } {
        Err(PhsError::invalid_argument("array was already released"))
    } else {
        Ok(())
    }
}

unsafe fn release_schema(schema: *mut ArrowSchema) {
    if let Some(release) = unsafe { (*schema).release } {
        unsafe { release(schema) };
    }
}

unsafe fn release_array(array: *mut ArrowArray) {
    if let Some(release) = unsafe { (*array).release } {
        unsafe { release(array) };
    }
}

unsafe fn take_array(array: *mut ArrowArray) -> PhsResult<ArrowArray> {
    unsafe { ensure_array_live(array)? };
    let value = unsafe { ptr::read(array) };
    unsafe { (*array).release = None };
    Ok(value)
}
```

```haskell
{- |
Module      : Polars.Arrow
Description : Arrow C Data Interface import helpers.

This module imports standard Arrow C Data Interface RecordBatches into managed
Polars DataFrames. The pointer wrapper is explicitly unsafe because pointer
validity and release callbacks come from the Arrow producer.
-}
module Polars.Arrow
    ( ArrowRecordBatch
    , fromArrowRecordBatch
    , unsafeArrowRecordBatch
    ) where

import Foreign.C.Types (CInt)
import Foreign.Marshal.Alloc (alloca)
import Foreign.Ptr (Ptr, castPtr, nullPtr)
import Foreign.Storable (peek, poke)

import Polars.DataFrame (DataFrame)
import Polars.Error (PolarsError)
import Polars.Internal.Managed (mkDataFrame)
import Polars.Internal.Raw (RawDataFrame, RawError, phs_dataframe_from_arrow_record_batch)
import Polars.Internal.Result (consumeError, nullPointerError)

data ArrowRecordBatch = ArrowRecordBatch !(Ptr ()) !(Ptr ())

unsafeArrowRecordBatch :: Ptr schema -> Ptr array -> ArrowRecordBatch
unsafeArrowRecordBatch schema array = ArrowRecordBatch (castPtr schema) (castPtr array)

fromArrowRecordBatch :: ArrowRecordBatch -> IO (Either PolarsError DataFrame)
fromArrowRecordBatch (ArrowRecordBatch schema array) = dataframeOut (phs_dataframe_from_arrow_record_batch schema array)

dataframeOut :: (Ptr (Ptr RawDataFrame) -> Ptr (Ptr RawError) -> IO CInt) -> IO (Either PolarsError DataFrame)
dataframeOut action =
    alloca $ \outPtr ->
        alloca $ \errPtr -> do
            poke outPtr nullPtr
            poke errPtr nullPtr
            status <- action outPtr errPtr
            if status == 0
                then do
                    ptr <- peek outPtr
                    if ptr == nullPtr
                        then pure (Left (nullPointerError "dataframe output"))
                        else Right <$> mkDataFrame ptr
                else Left <$> (consumeError status =<< peek errPtr)
```
